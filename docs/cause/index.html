Predicate: cause

Summary

(cause e1 e2)

Eventuality or agent e1 causes eventuality e2.

Characterizing Axioms

(forall (e1 e2)
  (if (cause e1 e2)
      (tcause e1 e2)))

A temporal entity may be an instant or an interval.

(forall (e1 e2)
  (if (cause e1 e2) (not (before e2 e1))))

Additional Axioms

(forall (a e2)
  (and (if (eventuality a)
           (iff (cause a e2) (cause0 a e2)))
       (if (agent a)
           (iff (cause a e2)
                (exists (e1)
                  (and (will' e1 a e2) (cause0 e1 e2)))))
       (if (not (or (eventuality a) (agent a)))
           (not (cause a e2)))))

(forall (a)
  (if (and (agent a) (etc))
      (exists (e) (cause a e))))

(forall (a e)
  (iff (agentOf a e)
       (and (agent a) (cause a e))))

(forall (x e)
  (iff (objectOf x e)
       (or (changeIn' e x)
           (exists (e1 e2)
             (and (and' e e1 e2) (cause e1 e2) (objectOf x e2))))))

(forall (y e)
  (iff (instrumentOf y e)
       (exists (a e1)
         (and (agentOf a e1) (changeIn' e1 y)
              (or (cause e1 e)
                  (exists (e2)
                    (and (cause e1 e2) (and' e e1 e2))))))))

(forall (e1 e2 s)
  (iff (enable0 e1 e2 s)
       (and (causalComplex s e2) (member e1 s)
            (not (cause e1 e2)))))

(forall (e1 e2)
  (iff (enable e1 e2)
       (forall (e3)
         (if (not' e3 e1)
             (exists (e4)
               (and (not' e4 e2) (cause e3 e4)))))))

(forall (e1 e2)
  (iff (allow e1 e2)
       (forall (e4)
         (if (not' e4 e2) (not (cause e1 e4))))))

(forall (e1 e2)
  (iff (prevent e1 e2)
       (exists (e4)
         (and (not' e4 e2) (cause e1 e4)))))

(forall (e1 e2)
  (iff (partiallyCause e1 e2)
       (exists (e3 e4)
         (and (not (cause e1 e2)) (not (cause e3 e2))
              (and' e4 e1 e3) (cause e4 e2)))))

(forall (e1 e2)
  (iff (tcause e1 e2)
       (exists (s c)
         (and (causalComplex s e2) (member e1 s)
              (deleteElt s1 s e1) (likely s1 c)
              (if (Rexist s)
                  (cause e1 e2))))))

(forall (e1 e2 q c)
  (iff (tcauseq e1 e2 q c)
       (exists (s)
         (and (causalComplex s e2) (member e1 s)
              (deleteElt s1 s e1) (likelihood q s1 c)
              (if (Rexist s1) (cause e1 e2))))))

(forall (s1 s a)
  (iff (evsBeyondControl s1 s a)
       (and (eventualities s) (subset s1 s) (agent a)
            (forall (e)
              (iff (member e s1)
                   (and (not (agentOf a e))
                        (not (exists (e1)
                               (and (agentOf a e1)
                                    (cause e1 e))))))))))

(forall (a e c)
  (iff (able a e c)
       (exists (s s1 e1)
         (and (causalComplex s e) (cause' e1 a e) (member e1 s)
              (evsBeyondControl s1 s a)
              (if (Rexist s1) (possible e1 c))))))

(forall (e1 e2)
  (iff (dcause e1 e2)
       (and (cause e1 e2)
            (not (exists (e3)
                   (and (cause e1 e3) (cause e3 e2)))))))

(forall (e1 a e)
  (if (and (will' e1 a e) (cause e1 e))
      (dcause e1 e)))

(forall (e a c t)
  (iff (executable e a c t)
       (exists (s)
         (and (enabled s e t)
              (or (exists (e1)
                    (and (dcause' e1 a e) (possible e1 c)))
                  (exists (e2)
                    (and (cause e2 e)
                         (executable e2 a c t))))))))

A temporal entity may be an instant or an interval.

(forall (e1 e2 t1)
  (if (and (cause e1 e2) (atTime e1 t1) (etc))
      (exists (t2)
