      (iff (patternParameters s x)
           (forall (y)
             (iff (member y s)
                  (exists (s1)
                         (and (componentOf y x)
                              (typelt y s1))))))))

(forall (x1 x f)
  (iff (xCoordinate x1 x f)
       (exists (v a1)
         (and (vertical2 v f) (member x v) (xAxis a1 f)
              (member x1 v) (member x1 a1)))))

(forall (y x)
  (iff (componentOf y x)
       (and (compositeEntity x)
            (exists (s)
              (and (componentsOf s x) (member y s))))))

(forall (y1 x f)
  (iff (yCoordinate y1 x f)
       (exists (h a2)
         (and (horizontal2 h f) (member x h) (yAxis a2 f)
              (member y1 h) (member y1 a2)))))

(forall (y1 y x1 x e s1 s2)
  (if (and (skfct y1 y x1 x e) (typelt x s1) (rangeFd s2 y x e))
      (exists (f)
        (and (function0 f s1 s2)
             (forall (p)
               (iff (pair p x1 y1) (member p f)))))))
