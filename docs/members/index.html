Predicate: members

Characterizing Axioms

(forall (g m)
  (if (members m g)
      (exists (e1)
        (member-nn e1 m))))

(forall (g m)
  (if (members m g)
      (exists (e1 e2)
        (and (member-nn e1 m) (of-in e2 m g)))))
