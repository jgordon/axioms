Predicate: inScale

Summary

(inScale y s)

y is a component of the scale s.

Characterizing Axioms

(forall (y s)
   (iff (inScale y s)
        (and (scale s) (componentOf y s))))

Additional Axioms

We can apply lts and gts either to points on the scale or to entities that are at those points.

(forall (x y s)
   (if (lts x y s)
       (and (or (inScale x s)
                (exists (x1) (and (at x x1 s) (inScale x1 s))))
            (or (inScale y s)
                (exists (y1) (and (at y y1 s) (inScale y1 s)))))))

(forall (x s)
   (iff (onScale x s)
        (or (inScale x s)
            (exists (y)
               (and (inScale y s) (at x y s))))))

(forall (x s)
   (iff (top x s)
        (and (inScale x s)
             (forall (y)
               (if (inScale y s) (leqs y x s))))))

(forall (x s)
   (iff (bottom x s)
        (and (inScale x s)
             (forall (y)
               (if (inScale y s) (leqs x y s))))))

(forall (s1 s)
   (iff (subscale s1 s)
        (and (scale s1) (scale s)
             (forall (x) (if (inScale x s1) (inScale x s)))
             (forall (x y)
                (iff (lts x y s1)
                     (and (inScale x s1) (inScale y s1)
                          (lts x y s)))))))

(forall (s1 s)
   (iff (reverse s1 s)
        (and (scale s1) (scale s)
             (forall (x y)
                (and (iff (inScale x s) (inScale x s1))
                     (iff (lts x y s) (lts y x s1)))))))

(forall (e x y s)
   (if (lts' e x y s)
       (iff (totalOrdering e x y s)
            (forall (x1 y1)
               (if (and (inScale x1 s) (inScale y1 s))
                   (or (lts x1 y1 s) (equal x1 y1)
                       (lts y1 x1 s)))))))

(forall (s s1 s2 x y)
   (if (and (Hi s1 s) (Lo s2 s) (inScale x s1) (inScale y s2))
       (lts y x s)))

Then the proportion of x to y is the fraction whose numerator and denominator are the numbers the measure maps x and y into, respectively.

(forall (f x y m s1)
  (if (and (measure m s1) (inScale x s1) (inScale y s1))
      (iff (proportion f x y m)
           (exists (n1 n2)
             (and (map m x n1) (map m y n2)
                  (fraction f n1 n2))))))

(forall (f x y m)
  (if (proportion f x y m)
      (exists (s1)
        (and (measure m s1) (inScale x s1) (inScale y s1)))))

(forall (x y s1 m)
  (if (sameHOM x y s1 m)
      (and (inScale x s1) (inScale y s1) (measure m s1))))

A set of events is frequent if it is in the Hi region of a rate scale.

(forall (s1)
  (iff (frequent s1)
       (exists (s s0 s2)
         (and (member s1 s0) (rateScale s s0) (Hi s2 s)
              (inScale s1 s2)))))

An eventuality is likely with respect to constraints c if its likelihood d is on the high end of the scale of likelihoods.

(forall (e c)
  (iff (likely e c)
       (exists (s d s1)
         (and (likelihood d e c) (likelihoodScale s) (Hi s1 s)
              (inScale d s1)))))
