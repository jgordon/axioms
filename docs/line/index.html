(line c)

(forall (c)
  (if (line c)
      (not (exists (x)

(forall (c)
  (if (linear c)
      (and (exists (c1) (and (line c1) (subfigure c c1)))
           (forall (c1 c2)
             (if (and (line c1) (line c2)
                      (subfigure c c1) (subfigure c c2))
                  (equal c1 c2))))))


(forall (c)
  (iff (linear c)
       (or (line c) (ray c) (lineSegment c))))


(forall (c1 c2)
  (if (and (line c1) (line c2))
      (iff (parallel c1 c2)
           (and (coplanar c1 c2)
                (not (exists (f)
                       (and (inside f c1) (inside f c2))))))))


(forall (f1 f2)
  (iff (parallel f1 f2)
       (exists (c1 c2)
         (and (line c1) (line c2) (inside f1 c1) (inside f2 c2)))))


(forall (r)

       (forall (c x1 x2)
